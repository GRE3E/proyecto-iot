// ========================================
// ESCLAVO 1 - LUCES
// ========================================
#include <Wire.h>

#define SLAVE_ADDRESS 8
#define MAX_CMD_LENGTH 32

const int ledPins[9] = {2, 3, 4, 5, 6, 7, 8, 9, 10};
bool estadoLuces[9] = {false};

void consolePrint(const char* message) {
  Serial.println(message);
}

void setup() {
  Serial.begin(115200);
  Wire.begin(SLAVE_ADDRESS);
  Wire.onReceive(receiveEvent);
  Wire.onRequest(requestEvent);
  
  for (int i = 0; i < 9; i++) {
    pinMode(ledPins[i], OUTPUT);
    digitalWrite(ledPins[i], LOW);
  }
  
  consolePrint(F("=== SLAVE 1 - Control de Luces"));
  consolePrint(F("Direccion I2C: 8"));
  consolePrint(F("9 luces disponibles (0-8)"));
  consolePrint(F("0=GARAJE, 1=PASILLO, 2=H.INVITADOS, 3=B.INVITADOS"));
  consolePrint(F("4=LAVANDERIA, 5=B.PRINCIPAL, 6=H.PRINCIPAL, 7=COCINA, 8=SALA"));
  consolePrint(F("Pines: 2-10"));
  consolePrint(F("Sistema listo"));
}

void loop() {
  delay(10);
}

void receiveEvent(int howMany) {
  char cmd[MAX_CMD_LENGTH + 1];
  int i = 0;
  
  while (Wire.available() && i < MAX_CMD_LENGTH) {
    cmd[i++] = Wire.read();
  }
  cmd[i] = '\0'; // Null-terminate the string
  
  // Clear any remaining bytes in the buffer
  while (Wire.available()) {
    Wire.read();
  }
  
  // Convert to uppercase (in-place)
  for (int j = 0; cmd[j]; j++) {
    cmd[j] = toupper(cmd[j]);
  }
  
  char* guion_ptr = strchr(cmd, '_');
  
  if (guion_ptr != NULL && guion_ptr != cmd && *(guion_ptr + 1) != '\0') {
    *guion_ptr = '\0'; // Split the string at the underscore
    int index = atoi(cmd);
    char* accion = guion_ptr + 1;
    
    char consoleBuffer[60];
    
    if (index >= 0 && index < 9) {
      if (strcmp(accion, "ENCENDER") == 0) {
        estadoLuces[index] = true;
        digitalWrite(ledPins[index], HIGH);
        sprintf(consoleBuffer, "Luz %d ENCENDIDA [Pin %d]", index, ledPins[index]);
        consolePrint(consoleBuffer);
      }
      else if (strcmp(accion, "APAGAR") == 0) {
        estadoLuces[index] = false;
        digitalWrite(ledPins[index], LOW);
        sprintf(consoleBuffer, "Luz %d APAGADA [Pin %d]", index, ledPins[index]);
        consolePrint(consoleBuffer);
      }
      else {
        sprintf(consoleBuffer, "ERROR: Accion desconocida: %s", accion);
        consolePrint(consoleBuffer);
      }
    } else {
      sprintf(consoleBuffer, "ERROR: Indice de luz invalido: %d (rango: 0-8)", index);
      consolePrint(consoleBuffer);
    }
  } else {
    consolePrint(F("ERROR: Formato de comando invalido"));
    consolePrint(F("Formato esperado: NUMERO_ACCION (ej: 3_ENCENDER)"));
  }
}

void requestEvent() {
  for (int i = 0; i < 9; i++) {
    Wire.write(estadoLuces[i] ? '1' : '0');
  }
  consolePrint(F("Estados enviados al Master"));
}
